# Match simple

match
$x isa person, has name $y;
end;

 ==>

QueryPipeline(
  Pipeline(
    QueryStage(
      ClauseMatch(
        Match,
        Whitespace
        Patterns(
          Pattern(
            Statement(
              StatementThing(
                Var, 
                Whitespace,
                ThingConstraintList(
                  ThingConstraint(
                    IsaConstraint(
                      Isa,
                      Whitespace,
                      TypeRef(Label)
                    ),
                  ),
                  Comma,
                  Whitespace,
                  ThingConstraint(
                    HasConstraint(
                      Has,
                      Whitespace,
                      TypeRef(Label),
                      Whitespace,
                      Var,
                    )
                  )
                )
              )
            )
          ),
          Semicolon
        )
      )
    ) 
    Whitespace,
    End,
    Semicolon
  )
)
